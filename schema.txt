=========== Schema NFT Item ===========
{
    name: {
        type: String,
        description: Name of NFT item
    },
    description: {
        type: String,
        description: Description for NFT item
    },
    type: {
        type: Number,
        description: Type of NFT item, use enums
    },
    image: {
        type: String,
        description: URL resources image of item
    },
    properties: {
        type: Object,
        description: Properties of NFT item
    }
} 

=========== Properties of Tank item ===========
{
    level: {
        type: String,
        description: Level of Tank
    },
    armor: {
        type: Number,
        description: Armor of Tank
    }
}

=========== Properties of Bullet item ===========
{
    damage: {
        type: Number,
        description: Damage of Bullet
    },
}
